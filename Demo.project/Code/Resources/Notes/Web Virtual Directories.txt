Open cmd as Administrator 
Copy below script inside the command prompt.
This will create application pool, application, assign the application to pool, and add all virtual directories. You may have to change the site id and the bindings port.

cd c:\windows\system32\inetsrv
appcmd add apppool /name:"Demo.Web" /managedRuntimeVersion:v4.0 /managedPipelineMode:Integrated
appcmd add site /name:Demo.Web /id:17 /physicalPath:C:\TFS\Retail\Code\Demo.Web /bindings:http/*:8017:
appcmd set site /site.name:Demo.Web /[path='/'].applicationPool:Demo.Web
appcmd add vdir /app.name:Demo.Web/ /path:/Client /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Web\Client
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Common /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Common.Web\Common
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Security /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Security.Web\Security
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Runtime /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Runtime.Web\Runtime
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Integration /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Integration.Web\Integration
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/VR_GenericData /physicalPath:C:\TFS\Vanrise\Code\Vanrise.GenericData.Web\VR_GenericData
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Demo_Module /physicalPath:C:\TFS\Demo.project\Code\Demo.Module.Web\Demo_Module
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/Demo_BestPractices /physicalPath:C:\TFS\Demo.project\Code\Demo.BestPractices.Web\Demo_BestPractices
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/VR_NumberingPlan /physicalPath:C:\TFS\Vanrise\Code\Vanrise.NumberingPlan.Web\VR_NumberingPlan
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/VR_Tools /physicalPath:C:\TFS\Demo.project\Code\Vanrise.Tools.Web\VR_Tools
appcmd add vdir /app.name:Demo.Web/ /path:/Client/Modules/VR_DevTools /physicalPath:C:\TFS\Vanrise\Code\Vanrise.DevTools.Web\VR_DevTools
