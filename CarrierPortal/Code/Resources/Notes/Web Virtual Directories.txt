Open cmd command as Administrator 
Copy below script inside the command prompt.
This will create application pool, application, assign the application to pool, and add all virtual directories. You may have to change the site id and the bindings port.

cd c:\windows\system32\inetsrv
appcmd add apppool /name:"CarrierPortal.Web" /managedRuntimeVersion:v4.0 /managedPipelineMode:Integrated
appcmd add site /name:CarrierPortal.Web /id:12 /physicalPath:C:\TFS\Retail\Code\CarrierPortal.Web /bindings:http/*:8012:
appcmd set site /site.name:CarrierPortal.Web /[path='/'].applicationPool:CarrierPortal.Web
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Web\Client
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/Common /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Common.Web\Common
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/Security /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Security.Web\Security
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/Runtime /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Runtime.Web\Runtime
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/VR_GenericData /physicalPath:C:\TFS\Vanrise\Code\Vanrise.GenericData.Web\VR_GenericData
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/Analytic /physicalPath:C:\TFS\Vanrise\Code\Vanrise.Analytic.Web\Analytic
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/ExcelConversion /physicalPath:C:\TFS\Vanrise\Code\Vanrise.ExcelConversion.Web\ExcelConversion
appcmd add vdir /app.name:CarrierPortal.Web/ /path:/Client/Modules/CP_SupplierPricelist /physicalPath:C:\TFS\CarrierPortal\Code\CP.SupplierPricelist.Web\CP_SupplierPricelist
